<!DOCTYPE html>
<html>
<head>
    <meta charset="utf-8">
    <title>Redirecting...</title>
    <script type="text/javascript">
        // 获取当前出错的路径，比如 /about
        var segmentCount = 1; // 如果你的站点在子路径下，例如 `username.github.io/repo`，这里可能需要调整
        var path = location.pathname;

        // 根据你的仓库结构，可能只需要 path.split('/').slice(segmentCount).join('/')
        var newPath = location.protocol + '//' + location.host + '/';

        // 重定向到首页，并把当前路径作为查询参数传递过去
        window.location.href = newPath + '?redirect=' + encodeURIComponent(path);
    </script>
    <noscript>
        <meta http-equiv="refresh" content="0; url=/" />
    </noscript>
</head>
<body>
    <p>Redirecting... If not, <a href="/">click here</a>.</p>
</body>
